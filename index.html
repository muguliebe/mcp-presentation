MCP Server 설정


import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';

const server = new Server({
  name: 'error-monitoring-server',
  version: '1.0.0',
});

// Resources 등록
server.setRequestHandler(ListResourcesRequestSchema, async () => {
  return {
    resources: [
      {
        uri: 'workspace://logs',
        name: 'Error Logs',
        description: 'Application error logs'
      }
    ]
  };
});

WebSocket 에러 감지


const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

wss.on('connection', (ws) => {
  ws.on('message', async (data) => {
    const errorData = JSON.parse(data);
    
    // MCP Sampling 호출
    const analysis = await mcpClient.sampling({
      messages: [
        {
          role: 'user',
          content: `에러 분석: ${errorData.message}`
        }
      ]
    });
    
    // 팀 알림 전송
    await sendNotification(analysis);
  });
});

MCP Resources 활용


// 워크스페이스 파일 접근
server.setRequestHandler(ReadResourceRequestSchema, async (request) => {
  const { uri } = request.params;
  
  if (uri === 'workspace://logs') {
    const logContent = await fs.readFile('./logs/error.log', 'utf8');
    return {
      contents: [
        {
          uri,
          mimeType: 'text/plain',
          text: logContent
        }
      ]
    };
  }
});

AI Sampling 구현


async function analyzeError(errorData) {
  const response = await mcpClient.sampling({
    messages: [
      {
        role: 'system',
        content: '당신은 에러 분석 전문가입니다.'
      },
      {
        role: 'user',
        content: `
에러 정보: ${errorData.message}
스택 트레이스: ${errorData.stack}
발생 시간: ${errorData.timestamp}

이 에러를 분석하고 해결책을 제안해주세요.
        `
      }
    ],
    maxTokens: 500
  });
  
  return response.content;
}
